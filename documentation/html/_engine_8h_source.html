19/**
20 * @class GameObject
21 * @brief Bazowa klasa dla wszystkich obiektów w scenie 3D
22 *
23 * Klasa GameObject stanowi fundament dla każdego obiektu,
24 * który ma być rysowany w przestrzeni 3D. Przechowuje informacje
25 * o pozycji, rotacji, skali i kolorze obiektu.
26 */
27class GameObject {
28public:
29 glm::vec3 pos;
30 glm::vec3 rot;
31 float scale;
32 glm::vec3 color;
33
34 GameObject() : pos(0), rot(0), scale(1.0f), color(1) {}
35 virtual ~GameObject() {}
36
37 virtual void draw() {
38 drawGeometry();
39 }
40 virtual void drawGeometry() = 0;
41};

43/**
44 * @class TexturedObject
45 * @brief Klasa reprezentująca obiekt z teksturą
46 *
47 * Rozszerza GameObject o obsługę tekstur. Umożliwia
48 * rysowanie obiektów 3D z nałożonymi teksturami.
49 */
50class TexturedObject : public GameObject {
51protected:
52 BitmapHandler* texture;
53public:
54 TexturedObject(BitmapHandler* tex) : texture(tex) {}
55
56 void draw() override {
57 if(texture) {
58 glEnable(GL_TEXTURE_2D);
59 glBindTexture(GL_TEXTURE_2D, texture->getTextureID());
60 glColor3f(1,1,1);
61 }
62
63 drawGeometry();
64
65 if(texture) {
66 glDisable(GL_TEXTURE_2D);
67 }
68 }
69};
