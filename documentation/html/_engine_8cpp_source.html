122void Engine::keyboard(unsigned char key, int x, int y) {
123 keys[key] = true;
124 if (key == 27) glutLeaveMainLoop();
125
126 if (key == 'l' || key == 'L') lightEnabled = !lightEnabled;
127 if (key == 'k' || key == 'K') smoothShading = !smoothShading;
128
129 if (key == '1') spawnCube();
130 if (key == '2') spawnPyramid();
131 if (key == '3') spawnSphere();
132 if (key == '4') spawnTeapot();
133 if (key == '5') spawnTorus();
134
135 if (key == 9) {
136 if (!objects.empty()) selectedIdx = (selectedIdx + 1) % objects.size();
137 }
138
139 if (key == ']') targetFPS += 5;
140 if (key == '[') targetFPS = std::max(5, targetFPS - 5);
141
142 if (key == 127) {
143 if (!objects.empty() && selectedIdx != -1) {
144
145 delete objects[selectedIdx];
146 objects.erase(objects.begin() + selectedIdx);
147
148 std::cout << "Obiekt usuniety (Deleted)" << std::endl;
149
150 if (objects.empty()) {
151 selectedIdx = -1;
152 } else {
153 if (selectedIdx >= objects.size()) {
154 selectedIdx = (int)objects.size() - 1;
155 }
156 }
157 }
158 }
159}

168void Engine::processInput() {
169 float camSpeed = 0.15f;
170 if (keys['w'] || keys['W']) camera->ProcessKeyboard(FORWARD, camSpeed);
171 if (keys['s'] || keys['S']) camera->ProcessKeyboard(BACKWARD, camSpeed);
172 if (keys['a'] || keys['A']) camera->ProcessKeyboard(LEFT, camSpeed);
173 if (keys['d'] || keys['D']) camera->ProcessKeyboard(RIGHT, camSpeed);
174
175 if (selectedIdx != -1 && selectedIdx < objects.size()) {
176 float objSpeed = 0.1f;
177 GameObject* obj = objects[selectedIdx];
178
179 glm::vec3 forward = glm::normalize(glm::vec3(camera->Front.x, 0, camera->Front.z));
180 glm::vec3 right = camera->Right;
181
182 if (specialKeys[GLUT_KEY_UP]) obj->pos += forward * objSpeed;
183 if (specialKeys[GLUT_KEY_DOWN]) obj->pos -= forward * objSpeed;
184 if (specialKeys[GLUT_KEY_LEFT]) obj->pos -= right * objSpeed;
185 if (specialKeys[GLUT_KEY_RIGHT]) obj->pos += right * objSpeed;
186
187 if (keys[32]) obj->pos.y += objSpeed;
188 if (keys['c'] || keys['C']) obj->pos.y -= objSpeed;
189
190 if (keys['r'] || keys['R']) obj->rot.y += 2.0f;
191 if (keys['t'] || keys['T']) obj->rot.x += 2.0f;
192 if (keys['y'] || keys['Y']) obj->rot.z += 2.0f;
193
194 if (keys['+'] || keys['=']) obj->scale += 0.05f;
195 if (keys['-'] || keys['_']) obj->scale -= 0.05f;
196 }
197}

199void Engine::render() {
200 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
201
202 if (lightEnabled) glEnable(GL_LIGHTING); else glDisable(GL_LIGHTING);
203 glShadeModel(smoothShading ? GL_SMOOTH : GL_FLAT);
204
205 glMatrixMode(GL_MODELVIEW);
206 glLoadMatrixf(glm::value_ptr(camera->GetViewMatrix()));
207
208 GLfloat lp[] = { 0.0f, 10.0f, 5.0f, 1.0f };
209 glLightfv(GL_LIGHT0, GL_POSITION, lp);
210
211 for (int i = 0; i < objects.size(); ++i) {
212 glPushMatrix();
213 GameObject* obj = objects[i];
214
215 glTranslatef(obj->pos.x, obj->pos.y, obj->pos.z);
216 glRotatef(obj->rot.x, 1, 0, 0);
217 glRotatef(obj->rot.y, 0, 1, 0);
218 glRotatef(obj->rot.z, 0, 0, 1);
219 glScalef(obj->scale, obj->scale, obj->scale);
220
221 glColor3f(obj->color.r, obj->color.g, obj->color.b);
222 obj->draw();
223
224 if (i == selectedIdx) {
225 glDisable(GL_LIGHTING);
226 glDisable(GL_TEXTURE_2D);
227 glEnable(GL_BLEND);
228 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
229
230 glColor4f(1.0f, 1.0f, 1.0f, 0.25f);
231
232 glDepthFunc(GL_LEQUAL);
233
234 obj->drawGeometry();
235
236 glDepthFunc(GL_LESS);
237 glDisable(GL_BLEND);
238 if (lightEnabled) glEnable(GL_LIGHTING);
239 }
240
241 glPopMatrix();
242 }
243
244 drawInterface();
245 glutSwapBuffers();
246}

277void Engine::drawInterface() {
278 glDisable(GL_LIGHTING);
279 glMatrixMode(GL_PROJECTION); glPushMatrix(); glLoadIdentity();
280 gluOrtho2D(0, windowWidth, 0, windowHeight);
281 glMatrixMode(GL_MODELVIEW); glPushMatrix(); glLoadIdentity();
282
283 glColor3f(1, 1, 1);
284 glRasterPos2i(10, windowHeight - 20);
285 std::string s = "FPS: " + std::to_string(targetFPS) +
286 " | Objects: " + std::to_string(objects.size()) +
287 " | Sel: " + (selectedIdx == -1 ? "None" : std::to_string(selectedIdx));
288 for (char c : s) glutBitmapCharacter(GLUT_BITMAP_HELVETICA_12, c);
289
290 glPopMatrix(); glMatrixMode(GL_PROJECTION); glPopMatrix(); glMatrixMode(GL_MODELVIEW);
291 if(lightEnabled) glEnable(GL_LIGHTING);
292}
